使用範例

輸入

>>> GLib.Variant('i', 1)

顯示

GLib.Variant('i', 1)

輸入

>>> GLib.Variant('()', ())

顯示

GLib.Variant('()', ())

輸入

>>> GLib.Variant('(s)', ('chewing',))

顯示

GLib.Variant('(s)', ('chewing',))

輸入

>>> args = GLib.Variant('(s)', ('chewing',))

輸入

>>> print(args)

顯示

('chewing',)

輸入

>>> fruit = GLib.Variant('(ss)', ('apple', 'banana'))

輸入

fruit

顯示

GLib.Variant('(ss)', ('apple', 'banana'))

輸入

>>> print(fruit)

顯示

('apple', 'banana')
